#include <math.h>
#include <iostream>
using namespace std;
int main() 
{
    float x, y, r;
    cout<<"\nВведите значение x - "; cin>>x;
    cout<<"\nВведите значение y - "; cin>>y;
    //решение уравнения по формуле
    r=((sin(x)+cos(y))/(cos(x)-sin(y)))*tan(x*y);
    cout<<"\nОтвет равен - "<<r;
    return 0;
}
