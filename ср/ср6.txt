//СР6(3) У прилавка магазина выстроилась очередь из п покупателей. Время об-служивания i-го покупателя равно ti(i = 1,…,п). Определить время Сi пребывания i-го покупателя в очереди.
#include <iostream>
using namespace std;

int main()
{
  int n; 
  cout << "\nВведите количество покупателей: "; cin >> n;

  int t[n]; 
  cout << "\nВведите время обслуживания каждого покупателя:";
//Присвоение времени к покупателям
  for (int i = 0; i < n; i++)
{
    cin >> t[i];
}

  int C[n];
  C[0] = 0; 
//расчет прибывания кажого покупателя
  for (int i = 1; i < n; i++)
{
    C[i] = C[i - 1] + t[i - 1]; 
}

  cout << "\nВремя пребывания каждого покупателя в очереди:";
//Вывод времени покупателей
  for (int i = 0; i < n; i++) 
{
    cout << "\nПокупатель " << i + 1 << ": " << C[i] << " минут";
}

  return 0;
}
